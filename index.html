import { WebR } from 'https://webr.r-wasm.org/latest/webr.mjs';


const webR = new WebR();

let webRReady = false;


const statusDot = document.getElementById('status-dot');

const statusText = document.getElementById('status-text');

const loadMsg = document.getElementById('load-msg');


(async () => {

  try {

    loadMsg.textContent = 'Downloading WebR (~5 MB)...';

    await webR.init();

    webRReady = true;

    statusDot.className = 'status-dot ready';

    statusText.textContent = 'R is ready';

    loadMsg.textContent = 'R engine ready!';

    setTimeout(() => {

      const ol = document.getElementById('loading-overlay');

      ol.classList.add('fade-out');

      setTimeout(() => ol.remove(), 500);

    }, 600);

  } catch(e) {

    statusDot.className = 'status-dot error';

    statusText.textContent = 'R failed to load';

    loadMsg.textContent = 'Error: ' + e.message;

  }

})();

// ─── Run Code ─────────────────────────────────────────────────

window.runCode = async function(lessonIdx, exIdx, isPlot) {

const code = document.getElementById(`code-${lessonIdx}-${exIdx}`).value;

const outEl = document.getElementById(`out-${lessonIdx}-${exIdx}`);

const plotEl = document.getElementById(`plot-${lessonIdx}-${exIdx}`);


if (!webRReady) {

outEl.textContent = ‘⏳ WebR is still loading. Please wait a moment…’;

outEl.className = ‘output-area visible loading-out’;

return;

}


outEl.textContent = ‘⏳ Running…’;

outEl.className = ‘output-area visible loading-out’;

plotEl.className = ‘output-plot’;


try {

let result;

if (isPlot) {

// Capture plot as PNG

await webR.evalRVoid(`png_file <- tempfile(fileext=".png") png(png_file, width=600, height=400, res=96) ${code} dev.off()`);

const pngData = await webR.evalR(`readBin(png_file, "raw", file.info(png_file)$size)`);

const arr = await pngData.toArray();

const uint8 = new Uint8Array(arr);

const blob = new Blob([uint8], { type: ‘image/png’ });

const url = URL.createObjectURL(blob);

plotEl.innerHTML = `<img src="${url}" alt="R plot">`;

plotEl.className = ‘output-plot visible’;

outEl.className = ‘output-area’;

} else {

// Capture text output

const shelter = await new webR.Shelter();

try {

const captured = await shelter.captureR(code, { withAutoprint: true });

const output = captured.output

.filter(o => o.type === ‘stdout’ || o.type === ‘stderr’)

.map(o => o.data)

.join(’\n’);

outEl.textContent = output || ‘(no output)’;

outEl.className = ‘output-area visible success’;

} finally {

shelter.purge();

}

}

} catch(e) {

outEl.textContent = ’❌ Error: ’ + e.message;

outEl.className = ‘output-area visible error-out’;

}

};

document.getElementById(`out-${lessonIdx}-${exIdx}`).className = ‘output-area’;

};


// ─── Quiz ─────────────────────────────────────────────────────

window.answerQuiz = function(lessonIdx, optIdx) {

const q = lessons[lessonIdx].quiz;

const opts = document.querySelectorAll(`[id^="qopt-${lessonIdx}-"]`);

const fb = document.getElementById(`qfb-${lessonIdx}`);

opts.forEach(o => o.style.pointerEvents = ‘none’);

opts[optIdx].classList.add(optIdx === q.correctIdx ? ‘correct’ : ‘wrong’);

opts[q.correctIdx].classList.add(‘correct’);

fb.className = ‘quiz-feedback show ’ + (optIdx === q.correctIdx ? ‘correct’ : ‘wrong’);

if (optIdx === q.correctIdx) {

completed.add(lessonIdx);

document.getElementById(`num-${lessonIdx}`).textContent = ‘✓’;

document.querySelectorAll(’.lesson-btn’)[lessonIdx].classList.add(‘done’);

}

};
